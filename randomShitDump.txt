#include <Arduino.h>
#include <LittleFS.h>

// 1. Set up the filesystem engine targeting the 16MB QSPI flash
LittleFS_Program QSPI_Drive; 

// 2. Define your gigantic data array size (Adjust this number as needed!)
const uint32_t ARRAY_SIZE = 1000; 

void setup() {
    Serial.begin(115200);
    while (!Serial); // Wait for you to open the serial monitor

    Serial.println("\n=== GIGA External Flash Asset Loader ===");

    // 3. Mount the 16MB flash drive
    if (!QSPI_Drive.begin()) {
        Serial.println("[-] Mount failed! You must run the format script once first.");
        while (1); 
    }
    Serial.println("[+] 16MB External Flash Mounted successfully.");

    // 4. Check if our array file is already saved on the chip
    if (!QSPI_Drive.exists("big_array.bin")) {
        Serial.println("[*] Array file not found. Creating it now...");

        // Allocate a temporary buffer in RAM just to generate your initial data
        uint8_t* tempBuffer = (uint8_t*)malloc(ARRAY_SIZE);
        if (tempBuffer == nullptr) {
            Serial.println("[-] RAM allocation failed for temporary buffer!");
            while(1);
        }

        // Fill the temporary buffer with mock data (e.g., 0, 1, 2, 3...)
        for (uint32_t i = 0; i < ARRAY_SIZE; i++) {
            tempBuffer[i] = (uint8_t)(i % 256);
        }

        // Open a fresh file on the 16MB flash chip
        File file = QSPI_Drive.open("big_array.bin", FILE_WRITE_BEGIN);
        if (file) {
            // Write the entire array from RAM straight into the physical flash chip
            file.write(tempBuffer, ARRAY_SIZE);
            file.close();
            Serial.println("[+] Success! Gigantic array safely baked into the 16MB Flash.");
        } else {
            Serial.println("[-] Failed to create file on Flash!");
        }

        // Clean up temporary RAM space immediately so it isn't wasted
        free(tempBuffer); 
    } else {
        Serial.println("[+] Array file detected! Skipping write process to preserve flash life.");
    }

    // 5. READ BACK TEST: Stream the data straight from the 16MB flash
    File readFile = QSPI_Drive.open("big_array.bin", FILE_READ);
    if (readFile) {
        Serial.print("[+] Verified File Size on Flash: ");
        Serial.print(readFile.size());
        Serial.println(" bytes.");

        Serial.println("[*] Printing first 20 bytes straight from the 16MB chip:");
        for (int i = 0; i < 20 && readFile.available(); i++) {
            uint8_t dataByte = readFile.read();
            Serial.print(dataByte);
            Serial.print(" ");
        }
        Serial.println("\n========================================");
        readFile.close();
    }
}

void loop() {
    // Your main loop stays empty and fast. Your array is safe on the chip.
}

//       );

// For now we use the default USB CDC backend defined in io.hpp.

void setup() {
    Serial.begin(115200);

    // Wait up to 2 s for a serial monitor (helpful during dev;
    // remove for headless / always-on deployment).
    for (int i = 0; i < 200 && !Serial; i++)
        delay(10);

    //start the ui manager (AwesomeUI)
    ui.basecolor = rgb(50, 50, 50);
    ui.begin();

    // Start the 1 ms tick thread.
    Scheduler::init();

    // Mount SD card (non-fatal — fs commands report errors if absent).
    if (!FS::init()) {
        IO::println("[warn] SD card not found - filesystem commands unavailable");
    }

    // Hand control to the shell (blocks forever).
    Shell::run();
}

void loop() {
    // loop() is never reached; Shell::run() owns the main thread.
    // mbed RTOS preempts background threads spawned via Scheduler::spawn().
}
